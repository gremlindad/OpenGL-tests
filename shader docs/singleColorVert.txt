#version 330 core


layout(location = 0) in vec3 aPos;

uniform mat4 model;


layout(std140) uniform Matrices
{
	uniform mat4 projection;
	uniform mat4 view;
};



void main()
{

	gl_Position = projection*view*model*vec4(aPos,1.0f);
	gl_PointSize = gl_Position.z;
	vs_out.TexCoords = vec2(1.0f,1.0f);
}
